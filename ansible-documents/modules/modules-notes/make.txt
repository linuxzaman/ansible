> MAKE    (/usr/lib/python2.7/dist-packages/ansible/modules/system/make.py)

        Run targets in a Makefile.

  * This module is maintained by The Ansible Community
OPTIONS (= is mandatory):

= chdir
        Change to this directory before running make.

        type: path

- file
        Use a custom Makefile.
        [Default: (null)]
        type: path
        version_added: 2.5

- params
        Any extra parameters to pass to make.
        [Default: (null)]
        type: dict

- target
        The target to run.
        Typically this would be something like `install',`test' or `all'."
        [Default: (null)]
        type: str


REQUIREMENTS:  make

EXAMPLES:

- name: Build the default target
  make:
    chdir: /home/ubuntu/cool-project

- name: Run 'install' target as root
  make:
    chdir: /home/ubuntu/cool-project
    target: install
  become: yes

- name: Build 'all' target with extra arguments
  make:
    chdir: /home/ubuntu/cool-project
    target: all
    params:
      NUM_THREADS: 4
      BACKEND: lapack

- name: Build 'all' target with a custom Makefile
  make:
    chdir: /home/ubuntu/cool-project
    target: all
    file: /some-project/Makefile
