> BIGIP_SERVICE_POLICY    (/usr/lib/python2.7/dist-packages/ansible/modules/network/f5/bigip_service_policy.py)

        Service policies allow you to configure timers and port misuse rules, if enabled, on a per rule or per context basis.

  * This module is maintained by an Ansible Partner
OPTIONS (= is mandatory):

- description

        [Default: (null)]
        type: str

        type: str

- partition
        Device partition to manage resources on.
        [Default: Common]
        type: str

        type: str

EXAMPLES:

- name: Create a service policy
  bigip_service_policy:
    name: foo
    timer_policy: timer1
    port_misuse_policy: misuse1
    timer_policy_enabled: yes
    port_misuse_policy_enabled: yes
    state: present
    provider:
      user: admin
      password: secret
      server: lb.mydomain.com
  delegate_to: localhost
