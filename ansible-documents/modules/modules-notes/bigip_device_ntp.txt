> BIGIP_DEVICE_NTP    (/usr/lib/python2.7/dist-packages/ansible/modules/network/f5/bigip_device_ntp.py)

        Manage NTP servers on a BIG-IP.

  * This module is maintained by an Ansible Partner

EXAMPLES:

- name: Set NTP server
  bigip_device_ntp:
    ntp_servers:
      - 192.0.2.23
    provider:
      password: secret
      server: lb.mydomain.com
      user: admin
  delegate_to: localhost

- name: Set timezone
  bigip_device_ntp:
    timezone: America/Los_Angeles
    provider:
      password: secret
      server: lb.mydomain.com
      user: admin
  delegate_to: localhost
